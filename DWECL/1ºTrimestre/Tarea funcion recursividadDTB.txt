        function pedirNumero(suma=0, contador=0){
            let valor1=(prompt("dame un valor: ")) // el prompt guarda siempre el valor como STRING
            if(valor1>0){
                suma+=parseInt(valor1);
                contador++;
                
                pedirNumero(suma,contador);
            }else{
                alert("la suma de estos valores es: "+suma);
                alert("la media de estos valores es: "+media(suma,contador));
            }
        }
        
        function media(a,b){
           media1 = a/b;
           return media1;
        }

        pedirNumero();
